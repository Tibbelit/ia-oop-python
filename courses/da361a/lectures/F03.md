class Person:

    pets = []
    persons = []

    def __init__(self, name):
        self.name = name
        self.persons.append(self)
        self.__age = 0

    def add_pet(self, pet):
        self.pets.append(pet)

    def get_age(self):
        return self.__age

    def set_age(self, age):
        if age.isdigit():
            __self.age = int(age)

    def __str__(self):
        return "{} har följande husdjur: {}".format(self.name, ", ".join(self.pets))

p1 = Person("Anton")
